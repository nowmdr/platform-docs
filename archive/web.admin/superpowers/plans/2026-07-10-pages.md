**Goal:** Раздел Pages в админке: список 7 статических страниц сайта + редактор SEO-полей (`pages`) и существующих hero-блоков (`hero_sections`) страницы.

**Architecture:** Спека — `docs/superpowers/specs/2026-07-10-pages-design.md` (утверждена). Паттерны целиком из blog/products (`docs/blog.md`, `docs/products.md` — прочитать перед началом): лёгкий список, форма RHF+Zod с раскладкой v2 (sticky-панель), гард несохранённых изменений `useBlocker`, контракт URL картинок (`resolveImageUrl`/`toStoragePath`), общий `ImagePickerDialog`. Упрощения против blog: create/delete НЕТ (набор страниц фиксирован), hero-секции редактируются только существующие (add/remove/reorder НЕТ → сохранение героев — только update по id, без diff insert/delete и без записи `position`).

**Tech Stack:** Vite + React 19 + TS, TanStack Query, RHF + Zod v4, shadcn/ui (все компоненты уже есть), Supabase через `getDb(site)`.

**Правила проекта (переопределяют шаблон плана):**
- Автотестов в репозитории нет — вместо TDD каждая задача завершается `npm run lint && npm run build` (build включает `tsc -b`); финальный ручной e2e-чек-лист — Task 5.
- **НЕ коммитить без явного разрешения пользователя** (CLAUDE.md) — в задачах нет шагов commit; в конце спросить разрешение.
- Текст UI — английский. `id` hero-секции в форме называется `sectionId` (RHF `useFieldArray` резервирует `id` под ключ рендера).

**Схему БД не менять.** Проверено по живой БД (2026-07-10): 7 строк `pages` (`home`,`shop`,`blog`,`terms`,`privacy`,`about`,`featured`), у `home`/`shop`/`blog` по 1 hero; `pages.meta` везде `{}` — НЕ читать и НЕ писать; `hero_sections.align` — check `left|right`; `updated_at` обеих таблиц — БД-триггерами.

```ts
import type { SiteConfig } from "@/config/sites";
import { getDb } from "@/lib/supabase";

// Строка pages (спека §8). Набор страниц фиксирован кодом сайта: create/delete
// из админки нет. Поле meta (jsonb) НЕ читаем и НЕ пишем — контракта с сайтом нет
// (решение 2026-07-10, спека раздела). slug read-only — на нём маршруты сайта.
export type Page = {
  id: string;
  created_at: string;
  updated_at: string;
  slug: string;
  seo_title: string | null;
  seo_description: string | null;
  og_image_path: string | null;
};

export type PageListItem = Pick<Page, "id" | "slug" | "seo_title"> & {
  heroCount: number;
};

// Payload update: редактируются только SEO-поля.
export type PageInput = Pick<Page, "seo_title" | "seo_description" | "og_image_path">;

// Hero-блок страницы. Редактируются только существующие секции (add/remove нет),
// поэтому id обязателен, а position не читаем и не пишем (порядок из БД).
export type HeroSection = {
  id: string;
  is_published: boolean;
  badge: string | null;
  title: string;
  description: string | null;
  bg_image_path: string | null;
  align: "left" | "right";
};

export type PageWithHeroes = { page: Page; heroes: HeroSection[] };

// Явный список колонок — чтобы не потащить meta через "*".
const PAGE_COLUMNS =
  "id,created_at,updated_at,slug,seo_title,seo_description,og_image_path";
const HERO_COLUMNS = "id,is_published,badge,title,description,bg_image_path,align";

export async function listPages(site: SiteConfig): Promise<PageListItem[]> {
  const db = getDb(site);
  const [pagesRes, heroesRes] = await Promise.all([
    db.from("pages").select("id,slug,seo_title").order("slug", { ascending: true }),
    db.from("hero_sections").select("page_id"),
  ]);
  if (pagesRes.error) throw pagesRes.error;
  if (heroesRes.error) throw heroesRes.error;

  const counts = new Map<string, number>();
  for (const row of (heroesRes.data ?? []) as { page_id: string }[]) {
    counts.set(row.page_id, (counts.get(row.page_id) ?? 0) + 1);
  }
  return ((pagesRes.data ?? []) as Pick<Page, "id" | "slug" | "seo_title">[]).map(
    (p) => ({ ...p, heroCount: counts.get(p.id) ?? 0 }),
  );
}

export async function getPage(
  site: SiteConfig,
  id: string,
): Promise<PageWithHeroes | null> {
  const db = getDb(site);
  const [pageRes, heroRes] = await Promise.all([
    db.from("pages").select(PAGE_COLUMNS).eq("id", id).maybeSingle(),
    db
      .from("hero_sections")
      .select(HERO_COLUMNS)
      .eq("page_id", id)
      .order("position", { ascending: true }),
  ]);
  if (pageRes.error) throw pageRes.error;
  if (heroRes.error) throw heroRes.error;
  if (!pageRes.data) return null;
  return {
    page: pageRes.data as Page,
    heroes: (heroRes.data ?? []) as HeroSection[],
  };
}

// Update страницы + update КАЖДОГО существующего hero по id (insert/delete нет —
// секции управляются миграциями сайта). Транзакции нет — осознанно (один админ);
// update идемпотентен, повторный Save после падения безопасен. Возвращает свежие
// данные для пересинхронизации формы (паттерн blog §1).
export async function updatePage(
  site: SiteConfig,
  id: string,
  input: PageInput,
  heroes: HeroSection[],
): Promise<PageWithHeroes> {
  const db = getDb(site);
  const { error } = await db.from("pages").update(input).eq("id", id);
  if (error) throw error;
  for (const { id: heroId, ...values } of heroes) {
    const { error: heroError } = await db
      .from("hero_sections")
      .update(values)
      .eq("id", heroId);
    if (heroError) throw heroError;
  }
  const fresh = await getPage(site, id);
  if (!fresh) throw new Error("Page not found after save");
  return fresh;
}
```

```tsx
import { useEffect, useState } from 'react'
import { useForm } from 'react-hook-form'
import { zodResolver } from '@hookform/resolvers/zod'
import { z } from 'zod'
import { Link, useBlocker, useOutletContext, useParams } from 'react-router-dom'
import { useMutation, useQuery, useQueryClient } from '@tanstack/react-query'
import { ArrowLeft, ImageOff, Images } from 'lucide-react'
import { toast } from 'sonner'
import type { SiteConfig } from '@/config/sites'
import {
  getPage,
  updatePage,
  type HeroSection,
  type PageInput,
  type PageWithHeroes,
} from '@/lib/pages'
import { resolveImageUrl, toStoragePath } from '@/lib/images'
import { ImagePickerDialog } from '@/features/products/ImagePickerDialog'
import { HeroSectionsEditor } from './HeroSectionsEditor'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { Textarea } from '@/components/ui/textarea'
import { Field, FieldGroup, FieldLabel, FieldError } from '@/components/ui/field'
import {
  AlertDialog,
  AlertDialogAction,
  AlertDialogCancel,
  AlertDialogContent,
  AlertDialogDescription,
  AlertDialogFooter,
  AlertDialogHeader,
  AlertDialogTitle,
} from '@/components/ui/alert-dialog'

// В форме поле id hero-секции называется sectionId: useFieldArray резервирует
// `id` под ключ рендера. Секции только существующие — sectionId обязателен.
const heroSchema = z.object({
  sectionId: z.string(),
  isPublished: z.boolean(),
  badge: z.string(),
  title: z.string().trim().min(1, 'Title is required'),
  description: z.string(),
  bgImagePath: z.string().trim(),
  align: z.literal('left').or(z.literal('right')),
})

const pageSchema = z.object({
  seo_title: z.string(),
  seo_description: z.string(),
  og_image_path: z.string().trim(),
  heroes: z.array(heroSchema),
})

export type PageFormValues = z.infer<typeof pageSchema>

const orNull = (v: string) => (v.trim() ? v.trim() : null)

// SEO-поля страницы — обычные optional (без механики Customize/Use defaults:
// у страницы нет собственных title/description для фолбэка). Пустое → null.
function toInput(site: SiteConfig, values: PageFormValues): PageInput {
  return {
    seo_title: orNull(values.seo_title),
    seo_description: orNull(values.seo_description),
    og_image_path: orNull(toStoragePath(site, values.og_image_path.trim())),
  }
}

function toHeroes(site: SiteConfig, values: PageFormValues): HeroSection[] {
  return values.heroes.map((h) => ({
    id: h.sectionId,
    is_published: h.isPublished,
    badge: orNull(h.badge),
    title: h.title.trim(),
    description: orNull(h.description),
    bg_image_path: orNull(toStoragePath(site, h.bgImagePath.trim())),
    align: h.align,
  }))
}

// В инпутах картинок показываем публичную ссылку (плоский ключ разворачиваем).
function toFormValues(site: SiteConfig, data: PageWithHeroes): PageFormValues {
  return {
    seo_title: data.page.seo_title ?? '',
    seo_description: data.page.seo_description ?? '',
    og_image_path: data.page.og_image_path
      ? resolveImageUrl(site, data.page.og_image_path)
      : '',
    heroes: data.heroes.map((h) => ({
      sectionId: h.id,
      isPublished: h.is_published,
      badge: h.badge ?? '',
      title: h.title,
      description: h.description ?? '',
      bgImagePath: h.bg_image_path ? resolveImageUrl(site, h.bg_image_path) : '',
      align: h.align,
    })),
  }
}

// Детальная страница: только edit (create/delete страниц из админки нет).
export function PageEditPage() {
  const site = useOutletContext<SiteConfig>()
  const { pageId } = useParams()

  const { data, isPending, error } = useQuery({
    queryKey: ['pages', site.slug, pageId],
    queryFn: () => getPage(site, pageId!),
  })

  return (
    <div className="flex flex-col gap-4">
      {(error || isPending || !data) && (
        <Link
          to={`/${site.slug}/pages`}
          className="flex w-fit items-center gap-1 text-sm text-muted-foreground transition-colors hover:text-foreground"
        >
          <ArrowLeft className="size-4" />
          Back to pages
        </Link>
      )}

      {error && (
        <p className="rounded-xl border border-destructive/50 p-4 text-sm text-destructive">
          Failed to load: {error.message}
        </p>
      )}

      {!error && isPending && <p className="text-sm text-muted-foreground">Loading…</p>}

      {!error && !isPending && !data && (
        <p className="rounded-xl border border-dashed p-8 text-center text-sm text-muted-foreground">
          Page not found.
        </p>
      )}

      {!error && !isPending && data && <PageForm site={site} data={data} />}
    </div>
  )
}

// Форма монтируется только с загруженными данными: defaultValues задаются один раз.
function PageForm({ site, data }: { site: SiteConfig; data: PageWithHeroes }) {
  const queryClient = useQueryClient()

  const form = useForm<PageFormValues>({
    resolver: zodResolver(pageSchema),
    defaultValues: toFormValues(site, data),
  })
  const { errors, isSubmitting, isDirty } = form.formState
  const ogPath = form.watch('og_image_path').trim()
  const [ogBroken, setOgBroken] = useState(false)
  const [pickerOpen, setPickerOpen] = useState(false)

  const save = useMutation({
    mutationFn: (values: PageFormValues) =>
      updatePage(site, data.page.id, toInput(site, values), toHeroes(site, values)),
    onSuccess: (fresh) => {
      // Пересинхронизация формы свежими данными (паттерн blog §1).
      form.reset(toFormValues(site, fresh))
      queryClient.setQueryData(['pages', site.slug, fresh.page.id], fresh)
      // exact: детальный ключ только что положен setQueryData — не сбрасывать.
      queryClient.invalidateQueries({ queryKey: ['pages', site.slug], exact: true })
      toast.success('Page saved')
    },
    onError: (e) => toast.error(e instanceof Error ? e.message : 'Failed to save page'),
  })

  // Гард несохранённых изменений (docs/products.md §4): !isPending обязателен.
  const blocker = useBlocker(isDirty && !save.isPending)

  function saveAndLeave() {
    void form.handleSubmit(
      async (values) => {
        try {
          await save.mutateAsync(values)
          blocker.proceed?.()
        } catch {
          blocker.reset?.()
        }
      },
      () => blocker.reset?.(),
    )()
  }

  useEffect(() => {
    if (!isDirty) return
    const warn = (e: BeforeUnloadEvent) => e.preventDefault()
    window.addEventListener('beforeunload', warn)
    return () => window.removeEventListener('beforeunload', warn)
  }, [isDirty])

  return (
    <div className="flex flex-col gap-4">
      {/* Sticky-панель действий: Save вне <form>, сабмит через form="page-form". */}
      <div className="sticky top-0 z-10 flex items-center gap-3 border-b bg-background py-2">
        <Link
          to={`/${site.slug}/pages`}
          className="flex w-fit items-center gap-1 text-sm text-muted-foreground transition-colors hover:text-foreground"
        >
          <ArrowLeft className="size-4" />
          Back to pages
        </Link>
        <div className="ml-auto flex items-center gap-3">
          <Button type="submit" form="page-form" disabled={isSubmitting || save.isPending}>
            {save.isPending ? 'Saving…' : 'Save'}
          </Button>
        </div>
      </div>

      <form
        id="page-form"
        onSubmit={form.handleSubmit(
          (values) => save.mutate(values),
          () => toast.error('Fix validation errors before saving'),
        )}
        noValidate
        className="flex max-w-3xl flex-col gap-6"
      >
        {/* Грид-шапка: превью OG image слева, SEO-поля справа. */}
        <div className="grid gap-6 md:grid-cols-[16rem_1fr]">
          <div>
            <div className="flex aspect-video items-center justify-center overflow-hidden rounded-xl border bg-muted/30 md:aspect-square">
              {ogPath && !ogBroken ? (
                <img
                  src={resolveImageUrl(site, ogPath)}
                  alt="OG image"
                  className="size-full object-cover"
                  onError={() => setOgBroken(true)}
                />
              ) : (
                <ImageOff className="size-8 text-muted-foreground" />
              )}
            </div>
            <p className="mt-2 truncate text-xs text-muted-foreground" title={data.page.slug}>
              Slug: {data.page.slug}
            </p>
          </div>

          <FieldGroup>
            <Field>
              <FieldLabel htmlFor="seo_title">SEO title</FieldLabel>
              <Input id="seo_title" {...form.register('seo_title')} />
            </Field>

            <Field>
              <FieldLabel htmlFor="seo_description">SEO description</FieldLabel>
              <Textarea id="seo_description" rows={3} {...form.register('seo_description')} />
            </Field>

            <Field data-invalid={!!errors.og_image_path}>
              <FieldLabel htmlFor="og_image_path">OG image</FieldLabel>
              <div className="flex gap-2">
                <Input
                  id="og_image_path"
                  placeholder="flat key in bucket or external URL"
                  {...form.register('og_image_path', {
                    onChange: () => setOgBroken(false),
                  })}
                />
                <Button type="button" variant="outline" onClick={() => setPickerOpen(true)}>
                  <Images />
                  Gallery
                </Button>
              </div>
              <FieldError errors={[errors.og_image_path]} />
            </Field>
          </FieldGroup>
        </div>

        <HeroSectionsEditor site={site} form={form} />
      </form>

      <ImagePickerDialog
        site={site}
        open={pickerOpen}
        onOpenChange={setPickerOpen}
        currentPath={ogPath ? toStoragePath(site, ogPath) : null}
        onSelect={(path) => {
          form.setValue('og_image_path', resolveImageUrl(site, path), {
            shouldDirty: true,
          })
          setOgBroken(false)
        }}
      />

      <AlertDialog
        open={blocker.state === 'blocked'}
        onOpenChange={(open) => {
          if (!open) blocker.reset?.()
        }}
      >
        <AlertDialogContent>
          <AlertDialogHeader>
            <AlertDialogTitle>Discard unsaved changes?</AlertDialogTitle>
            <AlertDialogDescription>
              You have unsaved changes. If you leave this page, they will be lost.
            </AlertDialogDescription>
          </AlertDialogHeader>
          <AlertDialogFooter>
            <AlertDialogCancel disabled={save.isPending}>Stay</AlertDialogCancel>
            <AlertDialogAction variant="destructive" onClick={() => blocker.proceed?.()}>
              Discard
            </AlertDialogAction>
            <Button type="button" onClick={saveAndLeave} disabled={save.isPending}>
              {save.isPending ? 'Saving…' : 'Save'}
            </Button>
          </AlertDialogFooter>
        </AlertDialogContent>
      </AlertDialog>
    </div>
  )
}
```

```tsx
import { useState } from 'react'
import {
  Controller,
  useFieldArray,
  type FieldError as RHFFieldError,
  type UseFormReturn,
} from 'react-hook-form'
import { Eye, EyeOff, Images } from 'lucide-react'
import type { SiteConfig } from '@/config/sites'
import { resolveImageUrl, toStoragePath } from '@/lib/images'
import { cn } from '@/lib/utils'
import { ImagePickerDialog } from '@/features/products/ImagePickerDialog'
import { Button } from '@/components/ui/button'
import { Input } from '@/components/ui/input'
import { Textarea } from '@/components/ui/textarea'
import { Field, FieldError, FieldLabel } from '@/components/ui/field'
import type { PageFormValues } from './PageEditPage'

// Hero-блоки страницы: редактируются только существующие секции — add/remove/
// reorder нет (набор управляется миграциями сайта), position не трогаем.
export function HeroSectionsEditor({
  site,
  form,
}: {
  site: SiteConfig
  form: UseFormReturn<PageFormValues>
}) {
  const { fields } = useFieldArray({ control: form.control, name: 'heroes' })
  // Индекс hero, для которого открыт пикер картинок; null — закрыт.
  const [pickerFor, setPickerFor] = useState<number | null>(null)

  const { errors } = form.formState

  return (
    <div className="flex flex-col gap-3">
      <span className="text-sm font-medium">Hero sections</span>

      {fields.length === 0 && (
        <p className="rounded-lg border border-dashed p-6 text-center text-sm text-muted-foreground">
          This page has no hero sections.
        </p>
      )}

      {fields.map((field, index) => {
        // watch в цикле: перерисовку гарантирует ре-рендер PageForm (не оборачивать в memo)
        const isPublished = form.watch(`heroes.${index}.isPublished`)
        const heroErrors = errors.heroes?.[index]
        return (
          <div
            key={field.id}
            className={cn('rounded-xl border', !isPublished && 'border-dashed opacity-60')}
          >
            <div className="flex items-center gap-2 border-b px-3 py-2">
              <span className="text-sm font-medium">Hero {index + 1}</span>
              {!isPublished && (
                <span className="text-xs text-muted-foreground">(hidden)</span>
              )}
              <div className="ml-auto flex items-center gap-1">
                <Button
                  type="button"
                  variant="ghost"
                  size="icon"
                  className="size-7"
                  aria-label={isPublished ? 'Hide section' : 'Show section'}
                  onClick={() =>
                    form.setValue(`heroes.${index}.isPublished`, !isPublished, {
                      shouldDirty: true,
                    })
                  }
                >
                  {isPublished ? <Eye className="size-4" /> : <EyeOff className="size-4" />}
                </Button>
              </div>
            </div>

            <div className="flex flex-col gap-3 p-3">
              <Field>
                <FieldLabel htmlFor={`hero-badge-${index}`}>Badge</FieldLabel>
                <Input
                  id={`hero-badge-${index}`}
                  placeholder="Optional"
                  {...form.register(`heroes.${index}.badge`)}
                />
              </Field>

              <Field data-invalid={!!heroErrors?.title}>
                <FieldLabel htmlFor={`hero-title-${index}`}>Title</FieldLabel>
                <Input
                  id={`hero-title-${index}`}
                  aria-invalid={!!heroErrors?.title}
                  {...form.register(`heroes.${index}.title`)}
                />
                <FieldError errors={[heroErrors?.title as RHFFieldError | undefined]} />
              </Field>

              <Field>
                <FieldLabel htmlFor={`hero-description-${index}`}>Description</FieldLabel>
                <Textarea
                  id={`hero-description-${index}`}
                  rows={3}
                  placeholder="Optional"
                  {...form.register(`heroes.${index}.description`)}
                />
              </Field>

              <Field>
                <FieldLabel htmlFor={`hero-bg-${index}`}>Background image</FieldLabel>
                <div className="flex gap-2">
                  <Input
                    id={`hero-bg-${index}`}
                    placeholder="flat key in bucket or external URL"
                    {...form.register(`heroes.${index}.bgImagePath`)}
                  />
                  <Button type="button" variant="outline" onClick={() => setPickerFor(index)}>
                    <Images />
                    Gallery
                  </Button>
                </div>
              </Field>

              <Controller
                control={form.control}
                name={`heroes.${index}.align`}
                render={({ field: alignField }) => (
                  <div className="flex items-center gap-1">
                    <span className="mr-1 text-sm text-muted-foreground">Text align:</span>
                    {(['left', 'right'] as const).map((side) => (
                      <Button
                        key={side}
                        type="button"
                        size="sm"
                        variant={alignField.value === side ? 'default' : 'outline'}
                        aria-pressed={alignField.value === side}
                        onClick={() => alignField.onChange(side)}
                      >
                        {side}
                      </Button>
                    ))}
                  </div>
                )}
              />
            </div>
          </div>
        )
      })}

      <ImagePickerDialog
        site={site}
        open={pickerFor !== null}
        onOpenChange={(open) => {
          if (!open) setPickerFor(null)
        }}
        currentPath={
          pickerFor !== null
            ? toStoragePath(site, form.getValues(`heroes.${pickerFor}.bgImagePath`).trim()) ||
              null
            : null
        }
        onSelect={(path) => {
          if (pickerFor === null) return
          form.setValue(`heroes.${pickerFor}.bgImagePath`, resolveImageUrl(site, path), {
            shouldDirty: true,
          })
        }}
      />
    </div>
  )
}
```
