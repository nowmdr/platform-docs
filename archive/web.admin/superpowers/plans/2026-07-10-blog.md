// Дата для списков/бейджей: "Jul 10, 2026" (язык интерфейса админки — английский).
export function formatDate(iso: string): string {
  return new Date(iso).toLocaleDateString("en-US", {
    year: "numeric",
    month: "short",
    day: "numeric",
  });
}

// timestamptz ISO → значение <input type="datetime-local"> в локальном времени.
export function toDatetimeLocalValue(iso: string): string {
  const d = new Date(iso);
  const pad = (n: number) => String(n).padStart(2, "0");
  return `${d.getFullYear()}-${pad(d.getMonth() + 1)}-${pad(d.getDate())}T${pad(
    d.getHours(),
  )}:${pad(d.getMinutes())}`;
}

import type { SiteConfig } from "@/config/sites";
import { getDb } from "@/lib/supabase";

// Строка posts (спека §8). slug автогенерируется БД из title, updated_at — триггером.
// Легаси-поле content НЕ читаем и НЕ пишем (решение 2026-07-10, спека блога §2).
export type Post = {
  id: string;
  created_at: string;
  updated_at: string;
  published_at: string;
  is_published: boolean;
  slug: string;
  title: string;
  excerpt: string | null;
  cover_image_path: string | null;
  seo_title: string | null;
  seo_description: string | null;
};

export type PostListItem = Pick<Post, "id" | "title" | "published_at" | "is_published">;

// Payload create/update: без id/timestamps/slug (slug генерирует БД).
export type PostInput = Omit<Post, "id" | "created_at" | "updated_at" | "slug">;

// Модуль тела поста. Порядок в массиве = position в БД: позиции СКВОЗНЫЕ между
// post_text_sections и post_product_sections — сайт сливает обе таблицы и
// сортирует по position. id отсутствует у ещё не сохранённых модулей.
export type PostModule =
  | {
      kind: "text";
      id?: string;
      isPublished: boolean;
      heading: string | null;
      body: string;
    }
  | {
      kind: "products";
      id?: string;
      isPublished: boolean;
      heading: string | null;
      columns: 2 | 3;
      productIds: string[];
    };

export type PostWithModules = { post: Post; modules: PostModule[] };

// Явный список колонок — чтобы случайно не потащить легаси content через "*".
const POST_COLUMNS =
  "id,created_at,updated_at,published_at,is_published,slug,title,excerpt,cover_image_path,seo_title,seo_description";

export async function listPosts(site: SiteConfig): Promise<PostListItem[]> {
  const { data, error } = await getDb(site)
    .from("posts")
    .select("id,title,published_at,is_published")
    // tiebreaker: даты импортированных постов могут совпадать (паттерн products)
    .order("published_at", { ascending: false })
    .order("title", { ascending: true });
  if (error) throw error;
  return (data ?? []) as PostListItem[];
}

type TextSectionRow = {
  id: string;
  position: number;
  is_published: boolean;
  heading: string | null;
  body: string;
};

type ProductSectionRow = {
  id: string;
  position: number;
  is_published: boolean;
  columns: 2 | 3;
  heading: string | null;
  product_ids: string[];
};

export async function getPost(
  site: SiteConfig,
  id: string,
): Promise<PostWithModules | null> {
  const db = getDb(site);
  const [postRes, textRes, productRes] = await Promise.all([
    db.from("posts").select(POST_COLUMNS).eq("id", id).maybeSingle(),
    db
      .from("post_text_sections")
      .select("id,position,is_published,heading,body")
      .eq("post_id", id),
    db
      .from("post_product_sections")
      .select("id,position,is_published,heading,columns,product_ids")
      .eq("post_id", id),
  ]);
  if (postRes.error) throw postRes.error;
  if (textRes.error) throw textRes.error;
  if (productRes.error) throw productRes.error;
  if (!postRes.data) return null;

  const withPosition: { position: number; module: PostModule }[] = [
    ...((textRes.data ?? []) as TextSectionRow[]).map((r) => ({
      position: r.position,
      module: {
        kind: "text" as const,
        id: r.id,
        isPublished: r.is_published,
        heading: r.heading,
        body: r.body,
      },
    })),
    ...((productRes.data ?? []) as ProductSectionRow[]).map((r) => ({
      position: r.position,
      module: {
        kind: "products" as const,
        id: r.id,
        isPublished: r.is_published,
        heading: r.heading,
        columns: r.columns,
        productIds: r.product_ids,
      },
    })),
  ];
  withPosition.sort((a, b) => a.position - b.position);

  return { post: postRes.data as Post, modules: withPosition.map((x) => x.module) };
}

// ---- Diff-сохранение секций (спека блога §3) ----

type SyncRow = { id?: string; values: Record<string, unknown> };

function toTextValues(
  postId: string,
  m: Extract<PostModule, { kind: "text" }>,
  position: number,
): Record<string, unknown> {
  return {
    post_id: postId,
    position,
    is_published: m.isPublished,
    heading: m.heading,
    body: m.body,
  };
}

function toProductValues(
  postId: string,
  m: Extract<PostModule, { kind: "products" }>,
  position: number,
): Record<string, unknown> {
  return {
    post_id: postId,
    position,
    is_published: m.isPublished,
    heading: m.heading,
    columns: m.columns,
    product_ids: m.productIds,
  };
}

// Одна таблица секций: insert новых (без id), update существующих (контент И
// position могли смениться — пишем всегда), delete пропавших — строго в конце.
async function syncTable(
  site: SiteConfig,
  table: "post_text_sections" | "post_product_sections",
  rows: SyncRow[],
  loadedIds: string[],
): Promise<void> {
  const db = getDb(site);
  const inserts = rows.filter((r) => !r.id).map((r) => r.values);
  const updates = rows.filter((r): r is { id: string; values: Record<string, unknown> } =>
    Boolean(r.id),
  );
  const keep = new Set(updates.map((r) => r.id));
  const deleteIds = loadedIds.filter((id) => !keep.has(id));

  if (inserts.length > 0) {
    const { error } = await db.from(table).insert(inserts);
    if (error) throw error;
  }
  for (const { id, values } of updates) {
    const { error } = await db.from(table).update(values).eq("id", id);
    if (error) throw error;
  }
  if (deleteIds.length > 0) {
    const { error } = await db.from(table).delete().in("id", deleteIds);
    if (error) throw error;
  }
}

async function syncModules(
  site: SiteConfig,
  postId: string,
  modules: PostModule[],
  loaded: PostModule[],
): Promise<void> {
  const textRows: SyncRow[] = [];
  const productRows: SyncRow[] = [];
  modules.forEach((m, position) => {
    if (m.kind === "text") {
      textRows.push({ id: m.id, values: toTextValues(postId, m, position) });
    } else {
      productRows.push({ id: m.id, values: toProductValues(postId, m, position) });
    }
  });
  const loadedTextIds = loaded
    .filter((m) => m.kind === "text" && m.id)
    .map((m) => m.id as string);
  const loadedProductIds = loaded
    .filter((m) => m.kind === "products" && m.id)
    .map((m) => m.id as string);
  await syncTable(site, "post_text_sections", textRows, loadedTextIds);
  await syncTable(site, "post_product_sections", productRows, loadedProductIds);
}

export async function createPost(
  site: SiteConfig,
  input: PostInput,
  modules: PostModule[],
): Promise<Post> {
  const db = getDb(site);
  // slug не отправляем — сгенерирует триггер posts_set_slug
  const { data, error } = await db.from("posts").insert(input).select(POST_COLUMNS).single();
  if (error) throw error;
  const post = data as Post;
  try {
    await syncModules(site, post.id, modules, []);
  } catch (e) {
    // откат (паттерн uploadImage): не оставляем пост-огрызок без секций
    await db.from("posts").delete().eq("id", post.id);
    throw e;
  }
  return post;
}

// Возвращает свежие пост+модули: у новых секций появились id, и форма ОБЯЗАНА
// пересинхронизироваться — иначе повторный Save продублирует новые модули.
export async function updatePost(
  site: SiteConfig,
  id: string,
  input: PostInput,
  modules: PostModule[],
  loaded: PostModule[],
): Promise<PostWithModules> {
  const { error } = await getDb(site).from("posts").update(input).eq("id", id);
  if (error) throw error;
  await syncModules(site, id, modules, loaded);
  const fresh = await getPost(site, id);
  if (!fresh) throw new Error("Post not found after save");
  return fresh;
}

export async function deletePost(site: SiteConfig, id: string): Promise<void> {
  // секции удалит FK ON DELETE CASCADE
  const { error } = await getDb(site).from("posts").delete().eq("id", id);
  if (error) throw error;
}

import { useDeferredValue, useState } from 'react'
import { Link, useOutletContext } from 'react-router-dom'
import { useQuery } from '@tanstack/react-query'
import { Loader2, Plus, Search, X } from 'lucide-react'
import type { SiteConfig } from '@/config/sites'

// Список постов блога: одна колонка строк-ссылок, поиск по заголовку (без фильтров).
export function PostsPage() {
  const site = useOutletContext<SiteConfig>()

  const [search, setSearch] = useState('')
  const deferredSearch = useDeferredValue(search)
  const isFiltering = deferredSearch !== search

  const { data: posts, isPending, error } = useQuery({
    queryKey: ['posts', site.slug],
    queryFn: () => listPosts(site),
  })

  const query = deferredSearch.trim().toLowerCase()
  const hasFilter = Boolean(query)
  const visible = (posts ?? []).filter(
    (p) => !query || p.title.toLowerCase().includes(query),
  )

  return (
    <div className="flex flex-col gap-4">
      <div className="flex items-center gap-3">
        <h1 className="text-lg font-semibold">Blog</h1>
        {posts && (
          <span className="text-sm text-muted-foreground">
            {hasFilter ? `${visible.length} / ${posts.length}` : posts.length}
          </span>
        )}
        <Button asChild className="ml-auto">
          <Link to="new">
            <Plus />
            New post
          </Link>
        </Button>
      </div>

      <div className="flex flex-wrap items-center gap-2">
        <div className="relative w-full max-w-56">
          {isFiltering ? (
            <Loader2 className="absolute top-1/2 left-2.5 size-4 -translate-y-1/2 animate-spin text-muted-foreground" />
          ) : (
            <Search className="absolute top-1/2 left-2.5 size-4 -translate-y-1/2 text-muted-foreground" />
          )}
          <Input
            type="search"
            placeholder="Search by title…"
            value={search}
            onChange={(e) => setSearch(e.target.value)}
            className="h-9 pl-8"
          />
        </div>
        {hasFilter && (
          <Button variant="ghost" className="h-9" onClick={() => setSearch('')}>
            <X />
            Clear
          </Button>
        )}
      </div>

      {error && (
        <p className="rounded-xl border border-destructive/50 p-4 text-sm text-destructive">
          Failed to load posts: {error.message}
        </p>
      )}

      {isPending && !error && (
        <div className="flex flex-col gap-1">
          {Array.from({ length: 8 }, (_, i) => (
            <Skeleton key={i} className="h-9 w-full rounded-md" />
          ))}
        </div>
      )}

      {posts && posts.length === 0 && (
        <p className="rounded-xl border border-dashed p-8 text-center text-sm text-muted-foreground">
          No posts yet.
        </p>
      )}

      {posts && posts.length > 0 && visible.length === 0 && (
        <p className="rounded-xl border border-dashed p-8 text-center text-sm text-muted-foreground">
          No posts match your search.
        </p>
      )}

      {visible.length > 0 && (
        <ul
          className={
            isFiltering
              ? 'flex flex-col opacity-60 transition-opacity'
              : 'flex flex-col transition-opacity'
          }
        >
          {visible.map((post) => (
            <li key={post.id}>
              <Link
                to={post.id}
                className="flex items-center justify-between gap-3 rounded-md px-3 py-2 transition-colors hover:bg-accent/50"
              >
                <span className="truncate text-sm font-medium">{post.title}</span>
                <span className="flex shrink-0 items-center gap-2 text-xs text-muted-foreground">
                  {!post.is_published && <Badge variant="outline">Draft</Badge>}
                  {formatDate(post.published_at)}
                </span>
              </Link>
            </li>
          ))}
        </ul>
      )}
    </div>
  )
}

  heading: z.string(),
  columns: z.literal(2).or(z.literal(3)),
  productIds: z.array(z.string()).min(1, 'Add at least one product'),
})

const postSchema = z.object({
  title: z.string().trim().min(1, 'Title is required'),
  excerpt: z.string(),
  cover_image_path: z.string().trim(),
  is_published: z.boolean(),
  published_at: z.string().min(1, 'Publish date is required'),
  seo_title: z.string(),
  seo_description: z.string(),
  modules: z.array(z.discriminatedUnion('kind', [textModuleSchema, productsModuleSchema])),
})

export type PostFormValues = z.infer<typeof postSchema>
export type ModuleFormValue = PostFormValues['modules'][number]

const orNull = (v: string) => (v.trim() ? v.trim() : null)

// Пустое или совпадающее с фолбэком SEO-значение не храним — фолбэк «живой»
// (паттерн products): при пустых полях сайт использует title/excerpt поста.
function seoOrNull(value: string, fallback: string): string | null {
  const trimmed = value.trim()
  return trimmed && trimmed !== fallback.trim() ? trimmed : null
}

function toInput(site: SiteConfig, values: PostFormValues): PostInput {
  return {
    title: values.title.trim(),
    excerpt: orNull(values.excerpt),
    cover_image_path: orNull(toStoragePath(site, values.cover_image_path.trim())),
    is_published: values.is_published,
    published_at: new Date(values.published_at).toISOString(),
    seo_title: seoOrNull(values.seo_title, values.title),
    seo_description: seoOrNull(values.seo_description, values.excerpt),
  }
}

function toModules(values: PostFormValues): PostModule[] {
  return values.modules.map((m) =>
    m.kind === 'text'
      ? {
          kind: 'text',
          id: m.sectionId,
          isPublished: m.isPublished,
          heading: orNull(m.heading),
          body: m.body,
        }
      : {
          kind: 'products',
          id: m.sectionId,
          isPublished: m.isPublished,
          heading: orNull(m.heading),
          columns: m.columns,
          productIds: m.productIds,
        },
  )
}

function toModuleFormValue(m: PostModule): ModuleFormValue {
  return m.kind === 'text'
    ? {
        kind: 'text',
        sectionId: m.id,
        isPublished: m.isPublished,
        heading: m.heading ?? '',
        body: m.body,
      }
    : {
        kind: 'products',
        sectionId: m.id,
        isPublished: m.isPublished,
        heading: m.heading ?? '',
        columns: m.columns,
        productIds: m.productIds,
      }
}

// В инпуте Cover image показываем публичную ссылку (плоский ключ разворачиваем).
function toFormValues(site: SiteConfig, data: PostWithModules | null): PostFormValues {
  return {
    title: data?.post.title ?? '',
    excerpt: data?.post.excerpt ?? '',
    cover_image_path: data?.post.cover_image_path
      ? resolveImageUrl(site, data.post.cover_image_path)
      : '',
    is_published: data?.post.is_published ?? true,

    seo_description: data?.post.seo_description ?? '',

  }
}

// Форма монтируется только с загруженными данными: defaultValues задаются один раз.
function PostForm({ site, data }: { site: SiteConfig; data: PostWithModules | null }) {
  const navigate = useNavigate()
  const queryClient = useQueryClient()
  const isNew = data === null
  const leavingRef = useRef(false) // Save из гард-модалки: не редиректить на созданный пост
  // Снапшот загруженных модулей для diff-сохранения; после Save обновляется
  // свежими данными (у новых секций появились id).
  const loadedModulesRef = useRef<PostModule[]>(data?.modules ?? [])

  const form = useForm<PostFormValues>({
    resolver: zodResolver(postSchema),
    defaultValues: toFormValues(site, data),
  })
  const { errors, isSubmitting, isDirty } = form.formState
  const coverPath = form.watch('cover_image_path').trim()
  const [coverBroken, setCoverBroken] = useState(false)
  const [pickerOpen, setPickerOpen] = useState(false)
  const [seoOpen, setSeoOpen] = useState(
    Boolean(data?.post.seo_title || data?.post.seo_description),
  )

  function openSeo() {
    if (!form.getValues('seo_title')) {
      form.setValue('seo_title', form.getValues('title'))
    }
    if (!form.getValues('seo_description')) {
      form.setValue('seo_description', form.getValues('excerpt'))
    }
    setSeoOpen(true)
  }

  function resetSeo() {
    form.setValue('seo_title', '', { shouldDirty: true })
    form.setValue('seo_description', '', { shouldDirty: true })
    setSeoOpen(false)
  }

  const save = useMutation({
    mutationFn: async (values: PostFormValues) => {
      const input = toInput(site, values)
      const modules = toModules(values)
      if (isNew) return { created: await createPost(site, input, modules), fresh: null }
      const fresh = await updatePost(
        site,
        data.post.id,
        input,
        modules,
        loadedModulesRef.current,
      )
      return { created: null, fresh }
    },
    onSuccess: ({ created, fresh }, values) => {
      if (fresh) {
        // Пересинхронизация: новые секции получили id; без reset свежими
        // значениями повторный Save продублировал бы их.
        loadedModulesRef.current = fresh.modules
        form.reset(toFormValues(site, fresh))
        queryClient.setQueryData(['posts', site.slug, fresh.post.id], fresh)
      } else {
        form.reset(values)
      }
      queryClient.invalidateQueries({ queryKey: ['posts', site.slug] })
      toast.success(isNew ? 'Post created' : 'Post saved')
      if (created && !leavingRef.current) {
        navigate(`/${site.slug}/blog/${created.id}`, { replace: true })
      }
    },
    onError: (e) => toast.error(e instanceof Error ? e.message : 'Failed to save post'),
  })

  const remove = useMutation({
    mutationFn: () => deletePost(site, data!.post.id),
    onSuccess: () => {
      form.reset()
      queryClient.invalidateQueries({ queryKey: ['posts', site.slug] })
      toast.success('Post deleted')
      navigate(`/${site.slug}/blog`, { replace: true })
    },
    onError: (e) => toast.error(e instanceof Error ? e.message : 'Failed to delete post'),
  })

  // Гард несохранённых изменений (docs/products.md §4): !isPending обязателен.
  const blocker = useBlocker(isDirty && !save.isPending && !remove.isPending)

  function saveAndLeave() {
    void form.handleSubmit(
      async (values) => {
        leavingRef.current = true
        try {
          await save.mutateAsync(values)
          blocker.proceed?.()
        } catch {
          blocker.reset?.()
        } finally {
          leavingRef.current = false
        }
      },
      () => blocker.reset?.(),
    )()
  }

  useEffect(() => {
    if (!isDirty) return
    const warn = (e: BeforeUnloadEvent) => e.preventDefault()
    window.addEventListener('beforeunload', warn)
    return () => window.removeEventListener('beforeunload', warn)
  }, [isDirty])

  return (
    <div className="flex flex-col gap-6 md:flex-row">
      <div className="w-full shrink-0 md:w-64">

          {coverPath && !coverBroken ? (
            <img
              src={resolveImageUrl(site, coverPath)}
              alt={form.watch('title') || 'Cover image'}
              className="size-full object-cover"
              onError={() => setCoverBroken(true)}
            />
          ) : (
            <ImageOff className="size-8 text-muted-foreground" />
          )}
        </div>
        {data && (
          <p className="mt-2 truncate text-xs text-muted-foreground" title={data.post.slug}>
            Slug: {data.post.slug}
          </p>
        )}
      </div>

      <form
        onSubmit={form.handleSubmit((values) => save.mutate(values))}
        noValidate
        className="min-w-0 max-w-xl flex-1"
      >
        <FieldGroup>
          <Field data-invalid={!!errors.title}>
            <FieldLabel htmlFor="title">Title</FieldLabel>
            <Input id="title" aria-invalid={!!errors.title} {...form.register('title')} />
            <FieldError errors={[errors.title]} />
          </Field>

          <Field>
            <FieldLabel htmlFor="excerpt">Excerpt</FieldLabel>
            <Textarea id="excerpt" rows={3} {...form.register('excerpt')} />
          </Field>

          <Field data-invalid={!!errors.cover_image_path}>
            <FieldLabel htmlFor="cover_image_path">Cover image</FieldLabel>
            <div className="flex gap-2">
              <Input
                id="cover_image_path"
                placeholder="flat key in bucket or external URL"
                {...form.register('cover_image_path', {
                  onChange: () => setCoverBroken(false),
                })}
              />
              <Button type="button" variant="outline" onClick={() => setPickerOpen(true)}>
                <Images />
                Gallery
              </Button>
            </div>
          </Field>

          <div className="grid grid-cols-2 items-end gap-4">
            <Field>
              <FieldLabel htmlFor="is_published">Status</FieldLabel>
              <Controller
                control={form.control}
                name="is_published"
                render={({ field }) => (
                  <div className="flex h-9 items-center gap-2">
                    <Switch
                      id="is_published"
                      checked={field.value}
                      onCheckedChange={field.onChange}
                    />
                    <span className="text-sm">{field.value ? 'Published' : 'Draft'}</span>
                  </div>
                )}
              />
            </Field>
            <Field data-invalid={!!errors.published_at}>
              <FieldLabel htmlFor="published_at">Published at</FieldLabel>
              <Input
                id="published_at"
                type="datetime-local"
                aria-invalid={!!errors.published_at}
                {...form.register('published_at')}
              />
              <FieldError errors={[errors.published_at]} />
            </Field>
          </div>

          {/* Content-модули — Task 5: <PostModulesEditor site={site} form={form} /> */}

          {!seoOpen ? (
            <Field>
              <FieldLabel>SEO</FieldLabel>
              <div className="flex items-center justify-between gap-3 rounded-lg border border-dashed p-3">
                <p className="text-sm text-muted-foreground">
                  Search engines use the post title and excerpt.
                </p>
                <Button type="button" variant="ghost" size="sm" onClick={openSeo}>
                  <Pencil />
                  Customize
                </Button>
              </div>
            </Field>
          ) : (

              <div className="flex items-center justify-between">
                <span className="text-sm font-medium">SEO</span>
                <Button type="button" variant="ghost" size="sm" onClick={resetSeo}>
                  <RotateCcw />
                  Use defaults
                </Button>
              </div>
              <Field>
                <FieldLabel htmlFor="seo_title">SEO title</FieldLabel>
                <Input id="seo_title" {...form.register('seo_title')} />
              </Field>
              <Field>
                <FieldLabel htmlFor="seo_description">SEO description</FieldLabel>
                <Textarea
                  id="seo_description"
                  rows={3}
                  {...form.register('seo_description')}
                />
              </Field>
            </>
          )}

          <div className="flex items-center gap-3">
            <Button type="submit" disabled={isSubmitting || save.isPending}>
              {save.isPending ? 'Saving…' : 'Save'}
            </Button>
            {!isNew && (
              <AlertDialog>
                <AlertDialogTrigger asChild>
                  <Button type="button" variant="destructive" disabled={remove.isPending}>
                    <Trash2 />
                    {remove.isPending ? 'Deleting…' : 'Delete'}
                  </Button>
                </AlertDialogTrigger>
                <AlertDialogContent>
                  <AlertDialogHeader>
                    <AlertDialogTitle>Delete this post?</AlertDialogTitle>
                    <AlertDialogDescription>
                      “{data.post.title}” and all its content sections will be
                      permanently deleted.
                    </AlertDialogDescription>
                  </AlertDialogHeader>
                  <AlertDialogFooter>
                    <AlertDialogCancel>Cancel</AlertDialogCancel>
                    <AlertDialogAction variant="destructive" onClick={() => remove.mutate()}>
                      Delete
                    </AlertDialogAction>
                  </AlertDialogFooter>
                </AlertDialogContent>
              </AlertDialog>
            )}
          </div>
        </FieldGroup>
      </form>

      <ImagePickerDialog
        site={site}
        open={pickerOpen}
        onOpenChange={setPickerOpen}
        currentPath={coverPath ? toStoragePath(site, coverPath) : null}
        onSelect={(path) => {
          form.setValue('cover_image_path', resolveImageUrl(site, path), {
            shouldDirty: true,
          })
          setCoverBroken(false)
        }}
      />

      <AlertDialog
        open={blocker.state === 'blocked'}
        onOpenChange={(open) => {
          if (!open) blocker.reset?.()
        }}
      >
        <AlertDialogContent>
          <AlertDialogHeader>
            <AlertDialogTitle>Discard unsaved changes?</AlertDialogTitle>
            <AlertDialogDescription>
              You have unsaved changes. If you leave this page, they will be lost.
            </AlertDialogDescription>
          </AlertDialogHeader>
          <AlertDialogFooter>
            <AlertDialogCancel disabled={save.isPending}>Stay</AlertDialogCancel>
            <AlertDialogAction variant="destructive" onClick={() => blocker.proceed?.()}>
              Discard
            </AlertDialogAction>
            <Button type="button" onClick={saveAndLeave} disabled={save.isPending}>
              {save.isPending ? 'Saving…' : 'Save'}
            </Button>
          </AlertDialogFooter>
        </AlertDialogContent>
      </AlertDialog>
    </div>
  )
}

      {fields.length === 0 && (
        <p className="rounded-lg border border-dashed p-6 text-center text-sm text-muted-foreground">
          No content yet. Add a text or products section below.
        </p>
      )}

      {fields.map((field, index) => {
        const isPublished = form.watch(`modules.${index}.isPublished`)
        const moduleErrors = errors.modules?.[index]
        return (
          <div
            key={field.id}
            className={cn(
              'rounded-xl border',
              !isPublished && 'border-dashed opacity-60',
            )}
          >
            <div className="flex items-center gap-2 border-b px-3 py-2">
              {field.kind === 'text' ? (
                <AlignLeft className="size-4 text-muted-foreground" />
              ) : (
                <LayoutGrid className="size-4 text-muted-foreground" />
              )}
              <span className="text-sm font-medium">
                {field.kind === 'text' ? 'Text' : 'Products'}
              </span>
              {!isPublished && (
                <span className="text-xs text-muted-foreground">(hidden)</span>
              )}
              <div className="ml-auto flex items-center gap-1">
                <Button
                  type="button"
                  variant="ghost"
                  size="icon"
                  className="size-7"
                  aria-label={isPublished ? 'Hide section' : 'Show section'}
                  onClick={() =>
                    form.setValue(`modules.${index}.isPublished`, !isPublished, {
                      shouldDirty: true,
                    })
                  }
                >
                  {isPublished ? <Eye className="size-4" /> : <EyeOff className="size-4" />}
                </Button>
                <Button
                  type="button"
                  variant="ghost"
                  size="icon"
                  className="size-7"
                  aria-label="Move up"
                  disabled={index === 0}
                  onClick={() => move(index, index - 1)}
                >
                  <ArrowUp className="size-4" />
                </Button>
                <Button
                  type="button"
                  variant="ghost"
                  size="icon"
                  className="size-7"
                  aria-label="Move down"
                  disabled={index === fields.length - 1}
                  onClick={() => move(index, index + 1)}
                >
                  <ArrowDown className="size-4" />
                </Button>
                <Button
                  type="button"
                  variant="ghost"
                  size="icon"
                  className="size-7 text-destructive hover:text-destructive"
                  aria-label="Remove section"
                  onClick={() => remove(index)}
                >
                  <Trash2 className="size-4" />
                </Button>
              </div>
            </div>

            <div className="flex flex-col gap-3 p-3">
              <Field>
                <FieldLabel htmlFor={`module-heading-${index}`}>Heading</FieldLabel>
                <Input
                  id={`module-heading-${index}`}
                  placeholder="Optional"
                  {...form.register(`modules.${index}.heading`)}
                />
              </Field>

              {field.kind === 'text' ? (
                <Field
                  data-invalid={
                    !!(moduleErrors && 'body' in moduleErrors && moduleErrors.body)
                  }
                >
                  <Controller
                    control={form.control}
                    name={`modules.${index}.body`}
                    render={({ field: bodyField }) => (
                      <RichTextEditor
                        value={bodyField.value}
                        onChange={bodyField.onChange}
                      />
                    )}
                  />
                  {moduleErrors && 'body' in moduleErrors && (
                    <FieldError errors={[moduleErrors.body]} />
                  )}
                </Field>
              ) : (
                <>
                  <Controller
                    control={form.control}
                    name={`modules.${index}.columns`}
                    render={({ field: columnsField }) => (
                      <div className="flex items-center gap-1">
                        <span className="mr-1 text-sm text-muted-foreground">Grid:</span>
                        {([2, 3] as const).map((n) => (
                          <Button
                            key={n}
                            type="button"
                            size="sm"
                            variant={columnsField.value === n ? 'default' : 'outline'}
                            onClick={() => columnsField.onChange(n)}
                          >
                            {n} columns
                          </Button>
                        ))}
                      </div>
                    )}
                  />

                  <Controller
                    control={form.control}
                    name={`modules.${index}.productIds`}
                    render={({ field: idsField }) => (
                      <div className="flex flex-col gap-1">
                        {idsField.value.length === 0 && (
                          <p className="rounded-lg border border-dashed p-4 text-center text-sm text-muted-foreground">
                            No products selected.
                          </p>
                        )}
                        {idsField.value.map((productId: string, i: number) => (
                          <ProductRow
                            key={productId}
                            site={site}
                            product={productById.get(productId)}
                            isFirst={i === 0}
                            isLast={i === idsField.value.length - 1}
                            onMoveUp={() => {
                              const next = [...idsField.value]
                              ;[next[i - 1], next[i]] = [next[i], next[i - 1]]
                              idsField.onChange(next)
                            }}
                            onMoveDown={() => {
                              const next = [...idsField.value]
                              ;[next[i], next[i + 1]] = [next[i + 1], next[i]]
                              idsField.onChange(next)
                            }}
                            onRemove={() =>
                              idsField.onChange(
                                idsField.value.filter((id: string) => id !== productId),
                              )
                            }
                          />
                        ))}
                        <Button
                          type="button"
                          variant="outline"
                          size="sm"
                          className="mt-1 w-fit"
                          onClick={() => setPickerFor(index)}
                        >
                          <Plus />
                          Add products
                        </Button>
                      </div>
                    )}
                  />
                  {moduleErrors && 'productIds' in moduleErrors && (
                    <FieldError errors={[moduleErrors.productIds]} />
                  )}
                </>
              )}
            </div>
          </div>
        )
      })}

      <div className="flex gap-2">
        <Button
          type="button"
          variant="outline"
          onClick={() =>
            append({ kind: 'text', isPublished: true, heading: '', body: '' })
          }
        >
          <Plus />
          Text
        </Button>
        <Button
          type="button"
          variant="outline"
          onClick={() =>
            append({
              kind: 'products',
              isPublished: true,
              heading: '',
              columns: 3,
              productIds: [],
            })
          }
        >
          <Plus />
          Products
        </Button>
      </div>

      <ProductPickerDialog
        site={site}
        open={pickerFor !== null}
        onOpenChange={(open) => {
          if (!open) setPickerFor(null)
        }}
        excludeIds={
          pickerFor !== null ? form.getValues(`modules.${pickerFor}.productIds`) : []
        }
        onAdd={(ids) => {
          if (pickerFor === null) return
          const current = form.getValues(`modules.${pickerFor}.productIds`)
          form.setValue(`modules.${pickerFor}.productIds`, [...current, ...ids], {
            shouldDirty: true,
            shouldValidate: true,
          })
        }}
      />
    </div>
  )
}

// Строка выбранного товара в продуктовом модуле; порядок строк = порядок
// карточек на сайте (product_ids). Товар мог быть удалён — показываем маркер.
function ProductRow({
  site,
  product,
  isFirst,
  isLast,
  onMoveUp,
  onMoveDown,
  onRemove,
}: {
  site: SiteConfig
  product: ProductListItem | undefined
  isFirst: boolean
  isLast: boolean
  onMoveUp: () => void
  onMoveDown: () => void
  onRemove: () => void
}) {
  return (
    <div className="flex items-center gap-2 rounded-md border px-2 py-1.5">
      {product?.image_path ? (
        <img
          src={resolveImageUrl(site, product.image_path)}
          alt={product.title}
          loading="lazy"
          className="size-8 shrink-0 rounded border object-cover"
        />
      ) : (
        <span className="flex size-8 shrink-0 items-center justify-center rounded border bg-muted/30">
          <ImageOff className="size-4 text-muted-foreground" />
        </span>
      )}
      <span className="min-w-0 flex-1 truncate text-sm">
        {product ? product.title : 'Unknown product (deleted?)'}
      </span>
      <Button
        type="button"
        variant="ghost"
        size="icon"
        className="size-6"
        aria-label="Move product up"
        disabled={isFirst}
        onClick={onMoveUp}
      >
        <ArrowUp className="size-3.5" />
      </Button>
      <Button
        type="button"
        variant="ghost"
        size="icon"
        className="size-6"
        aria-label="Move product down"
        disabled={isLast}
        onClick={onMoveDown}
      >
        <ArrowDown className="size-3.5" />
      </Button>
      <Button
        type="button"
        variant="ghost"
        size="icon"
        className="size-6 text-muted-foreground hover:text-destructive"
        aria-label="Remove product"
        onClick={onRemove}
      >
        <X className="size-3.5" />
      </Button>
    </div>
  )
}
